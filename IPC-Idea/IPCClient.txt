IPCClient:
	onRawMessageCome:
		decode -> IPCMessage:msg
		msg.wellformed:
		case msg.opCode:
			Reg:
			Req:
			Rej:
			Set:
			Get:
			Unreg:
				reqMsg = constructRequestMsg(msg.opCode, msg.opID, msg.address, payload = {})
				sender.send(reqMsg.toBytes())
				break
				
			ResRslt:
			ResErr:
				component.postMessage(msg)
				break
			Invalid:
			default:
				printLog()
				break;
